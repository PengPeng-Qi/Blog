# Date

# 创建日期

// new Date(year, month, date, hour, minute, second, millisecond)
let d1 = new Date(2012, 1, 20, 3, 12);
alert( d1 );  // Mon Feb 20 2012 03:12:00 GMT+0800 (中国标准时间)

// new Date(datastring)
let d2 = new Date("February 20, 2012 03:12:00"); 
alert( d2 ); // Mon Feb 20 2012 03:12:00 GMT+0800 (中国标准时间)

# getTime

返回日期的时间戳

let date = new Date();

date.getTime();  // 1639989223492

# 设置日期组件

  • setFullYear(year, [month], [date])
  • setMonth(month, [date])
  • setDate(date)
  • setHours(hour, [min], [sec], [ms])
  • setMinutes(min, [sec], [ms])
  • setSeconds(sec, [ms])
  • setMilliseconds(ms)
  • setTime(milliseconds)
let today = new Date();

today.setHours(0);
alert(today); // 日期依然是今天，但是小时数被改为了 0

today.setHours(0, 0, 0, 0);
alert(today); // 日期依然是今天，时间为 00:00:00。

# 自动校准

Date 对象是一个非常方便的特性，可以设置超范围的数值，它会自动校准

let date1 = new Date(2013, 0, 32); // 32 Jan 2013 ?!?
alert(date1); // ……是 1st Feb 2013!

let date2 = new Date(2016, 1, 28);
date2.setDate(date2.getDate() + 2);

alert( date2 ); // 1 Mar 2016

# 日期转化为数字

当Date 对象被转化为数字时，得到的是对应的时间戳，与使用date.getTime() 的结果相同：

let date = new Date();
alert(+date); // 1639990208973，与使用 date.getTime() 的结果相同

重要的副作用：

let start = new Date(); // 开始测量时间

// do the job
for (let i = 0; i < 100000; i++) {
  let doSomething = i * i * i;
}

let end = new Date(); // 结束测量时间
alert( `The loop took ${end - start} ms` );  // 3ms

# Date.now()

Date.now() 会返回当前的时间戳。相当于new Date().getTime()，但是不会创建中间的Date 对象。因此更快，且不会对垃圾处理造成额外的压力。

# 度量

// 我们有 date1 和 date2，哪个函数会更快地返回两者的时间差？
function diffSubtract(date1, date2) {
  return date2 - date1;
}

// or
function diffGetTime(date1, date2) {
  return date2.getTime() - date1.getTime();
}

这两个函数做的事情完全相同，后者使用显性date.getTime() 来获取毫秒形式的日期，前置依赖日期 - 数字 的转换来实现。

结论：使用getTime() 方式更快，因为没有进行类型转化，这样对引擎优化来说更加简单。

# Date.parse

Date.parse(str) 方法可以从一个字符串中读取日期。

字符串的格式应该为：YYYY-MM-DD HH:mm:ss:sss

Date.parse(str) 调用会解析给定格式的字符串，并返回时间戳，如果给定的字符串的格式不正确，则返回NaN

let ms = Date.parse('2012-01-26T13:51:50.417-07:00');

alert(ms); // 1327611110417  (时间戳)

我们可以通过时间戳来生成一个new Date 对象：

let date = new Date( Date.parse('2012-01-26T13:51:50.417-07:00') );

alert(date);

# 某月的最后一天

function getLastDayOfMonth(year, month) {
  /* 上一月的最后一天 */
  let date = new Date(year, month + 1, 0);
  return date.getDate();
}

alert( getLastDayOfMonth(2012, 0) ); // 31
alert( getLastDayOfMonth(2012, 1) ); // 29
alert( getLastDayOfMonth(2013, 1) ); // 28
